020package net.sf.jkniv.whinstone.jdbc;
021
022import java.sql.Connection;
023import java.sql.DriverManager;
024import java.sql.SQLException;
025import java.util.Properties;
026
027import net.sf.jkniv.sqlegance.ConnectionAdapter;
028import net.sf.jkniv.sqlegance.RepositoryProperty;
029import net.sf.jkniv.sqlegance.transaction.Isolation;
030
031public class DriverManagerAdapter extends AbstractJdbcAdapter
032{
033    private Properties                       props;
034    private Isolation                        defaultIsolation;
035    private String                           driver;
036    private String                           url;
037    
038    public DriverManagerAdapter(Properties props, String contextName)//(Properties props, Isolation defaultIsolation, String name)
039    {
040        super(contextName);
041        this.props = props;
042        this.defaultIsolation = Isolation.DEFAULT;
043        this.driver = props.getProperty(RepositoryProperty.JDBC_DRIVER.key());
044        this.url = props.getProperty(RepositoryProperty.JDBC_URL.key());
045    }
046    
047    /**
048     * Attempts to establish a connection to the database
049     * @return a Connection to the URL
050     * @throws net.sf.jkniv.sqlegance.RepositoryException if cannot establish a connection
051     */
052    public ConnectionAdapter open() 
053    {
054        return open(defaultIsolation);
055    }
056    
057    /**
058     * Attempts to establish a connection to the database with specific isolation 
059     * @param isolation transaction level for connection
060     * @return a Connection to the URL
061     * @throws net.sf.jkniv.sqlegance.RepositoryException if cannot establish a connection
062     */
063    public ConnectionAdapter open(Isolation isolation)
064    {
065        ConnectionAdapter adapter = null;
066        // TODO application must manually load any JDBC drivers prior to version 4.0
067        //if (driver != null && !drivers.contains(driver))
068        //    JdbcUtils.registerDriver(driver);
069        
070        try
071        {
072            LOG.debug("Getting Connection from DriverManager");
073            Connection conn = DriverManager.getConnection(url, props);
074            setIsolation(conn, isolation);
075            adapter = new JdbcConnectionAdapter(conn);
076        }
077        catch (SQLException e)
078        {
079            handlerException.handle(e,
080                    "SEVERE FAIL, cannot get database connection url [" + url + "] config=" + props);
081        }
082        return adapter;
083    }    
084}
