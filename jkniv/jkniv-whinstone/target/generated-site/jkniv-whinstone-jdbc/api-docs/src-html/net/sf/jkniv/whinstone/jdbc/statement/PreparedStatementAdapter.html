109    @Override
110    public StatementAdapter<T, ResultSet> bind(String name, Object value)
111    {
112        log(name, value);
113        try
114        {
115            if (name.toLowerCase().startsWith("in:"))
116            {
117                setValue((Object[]) value);
118            }
119            else if (value instanceof java.util.Date)
120            {
121                setValue((java.util.Date) value);
122            }
123            else if (Enum.class.isInstance(value))
124            {
125                setValue((Enum) value);
126            }
127            else if (value instanceof java.util.Calendar)
128            {
129                setValue((Calendar) value);
130            }
131            else
132            {
133                setValue(value);
134            }
135        }
136        catch (SQLException e)
137        {
138            this.handlerException.handle(e);// FIXME handler default message with custom params
139        }
140        return this;
141    }

186    @SuppressWarnings({ "rawtypes", "unchecked" })
187    public List<T> rows()
188    {
189        ResultSet rs = null;
190        ResultSetParser<T, ResultSet> rsParser = null;
191        Groupable<T, ?> grouping = new NoGroupingBy<T, T>();
192        List<T> list = Collections.emptyList();
193        try
194        {
195            rs = stmt.executeQuery();
196            
197            JdbcColumn<ResultSet>[] columns = getJdbcColumns(rs.getMetaData());
198            setResultRow(columns);
199            
200            Transformable<T> transformable = resultRow.getTransformable();
201            if (!groupingBy.isEmpty())
202            {
203                grouping = new GroupingBy(groupingBy, returnType, transformable);
204            }
205            rsParser = new ObjectResultSetParser(resultRow, grouping);
206            list = rsParser.parser(rs);
207        }
208        catch (SQLException e)
209        {
210            handlerException.handle(e, e.getMessage());
211        }
212        return list;
213    }

336    @SuppressWarnings({ "unchecked", "rawtypes" })
337    private void setResultRow(JdbcColumn<ResultSet>[] columns)
338    {
339        if (resultRow != null)
340            return;
341        
342        if (scalar)
343        {
344            resultRow = new ScalarResultRow(columns, sqlLogger);
345        }
346        else if (Map.class.isAssignableFrom(returnType))
347        {
348            resultRow = new MapResultRow(returnType, columns, sqlLogger);
349        }
350        else if (Number.class.isAssignableFrom(returnType)) // FIXME implements for date, calendar, boolean improve design
351        {
352            resultRow = new NumberResultRow(returnType, columns, sqlLogger);
353        }
354        else if (String.class.isAssignableFrom(returnType))
355        {
356            resultRow = new StringResultRow(columns, sqlLogger);
357        }
358        else if (oneToManies.isEmpty())
359        {
360            resultRow = new FlatObjectResultRow(returnType, columns, sqlLogger);
361        }
362        else
363        {
364            resultRow = new PojoResultRow(returnType, columns, oneToManies, sqlLogger);
365        }
366    }

369    {

373    private void log(String name, Object value)
374    {
375            if (sqlLogger.isEnabled(LogLevel.STMT))
376                sqlLogger.log(LogLevel.STMT,
377                        "Setting SQL Parameter from index [{}] with name [{}] with value of [{}] type of [{}]", index,
378                        name, sqlLogger.mask(name, value), (value == null ? "NULL" : value.getClass()));
379    }

382    {
383            String name = String.valueOf(index+indexIN);
384            if (sqlLogger.isEnabled(LogLevel.STMT))
385                sqlLogger.log(LogLevel.STMT,
386                        "Setting SQL Parameter from index [{}] with name [{}] with value of [{}] type of [{}]", index,
387                        name, sqlLogger.mask(name, value), (value == null ? "NULL" : value.getClass()));
388    }

390    /**
391     * Summarize the columns from SQL result in binary data or not.
392     * @param metadata  object that contains information about the types and properties of the columns in a <code>ResultSet</code> 
393     * @return Array of columns with name and index
394     * @throws SQLException Errors that occurs when access {@code ResultSetMetaData} methods.
395     */
396    private JdbcColumn[] getJdbcColumns(ResultSetMetaData metadata) throws SQLException
397    {
398        JdbcColumn[] columns = new JdbcColumn[metadata.getColumnCount()];
399        
400        for (int i = 0; i < columns.length; i++)
401        {
402            int columnNumber = i + 1;
403            String columnName = getColumnName(metadata, columnNumber);
404            int columnType = metadata.getColumnType(columnNumber);
405            //boolean binaryData = false;
406            //if (columnType == Types.CLOB || columnType == Types.BLOB)
407            //    binaryData = true;
408            columns[i] = new DefaultJdbcColumn(columnNumber, columnName, columnType);
409        }
410        return columns;
411    }
412    
413    private String getColumnName(ResultSetMetaData metaData, int columnIndex) throws SQLException
414    {
415        try
416        {
417            return metaData.getColumnLabel(columnIndex);
418        }
419        catch (SQLException e)
420        {
421            return metaData.getColumnName(columnIndex);
422        }
423    }
