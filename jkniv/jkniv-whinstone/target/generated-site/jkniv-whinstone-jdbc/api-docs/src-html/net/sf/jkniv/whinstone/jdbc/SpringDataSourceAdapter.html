020package net.sf.jkniv.whinstone.jdbc;
021
022import java.lang.reflect.Method;
023import java.sql.Connection;
024import java.sql.SQLException;
025
026import javax.sql.DataSource;
027
028import net.sf.jkniv.reflect.beans.ObjectProxy;
029import net.sf.jkniv.reflect.beans.ObjectProxyFactory;
030import net.sf.jkniv.sqlegance.ConnectionAdapter;
031import net.sf.jkniv.sqlegance.RepositoryException;
032import net.sf.jkniv.sqlegance.transaction.Isolation;
033
034@SuppressWarnings("unchecked")
035public class SpringDataSourceAdapter extends AbstractJdbcAdapter
036{
037    private static final String SPRING_TRANSACTIONAWEREDATASOURCE_PROXY = "org.springframework.jdbc.datasource.TransactionAwareDataSourceProxy";
038    private static final String SPRING_DATASOURCE_UTILS                 = "org.springframework.jdbc.datasource.DataSourceUtils";
039    private DataSource          dataSourceTarget;
040    private ObjectProxy<?>      transactionAwareDataSourceProxy;
041    private Isolation           defaultIsolation;
042    //TransactionAwareDataSourceProxy dsProxy;
043    
044    public SpringDataSourceAdapter(DataSource ds, String contextName)//DataSource dataSource, Isolation defaultIsolation, String name)
045    {
046        super(contextName);
047        this.defaultIsolation = Isolation.DEFAULT;
048        this.dataSourceTarget = ds;
049        //this.dsProxy = new TransactionAwareDataSourceProxy(ds);
050        
051        this.transactionAwareDataSourceProxy = ObjectProxyFactory
052                .newProxy(classForName(SPRING_TRANSACTIONAWEREDATASOURCE_PROXY));
053        this.transactionAwareDataSourceProxy.setConstructorArgs(dataSourceTarget);
054        this.transactionAwareDataSourceProxy.newInstance();
055    }
056    
057    public ConnectionAdapter open()
058    {
059        return open(defaultIsolation);
060    }
061    
062    public ConnectionAdapter open(Isolation isolation)
063    {
064        ConnectionAdapter adapter = null;
065        try
066        {
067            LOG.debug("Getting Connection from Spring TransactionAwareDataSourceProxy class");
068            Connection conn = (Connection) this.transactionAwareDataSourceProxy.invoke("getConnection");
069            conn.setAutoCommit(false);
070            adapter = new JdbcConnectionAdapter(conn);
071        }
072        catch (SQLException e)
073        {
074            handlerException.handle(e, "SEVERE FAIL, cannot get database connection datasource with Reason: " + e.getMessage());
075        }
076        return adapter;
077    }
078    
079    @Override
080    public void close(ConnectionAdapter conn)
081    {
082        LOG.debug("Release Connection [{}] from Spring DataSourceProxy", conn);
083        doReleaseConnection(conn);
084    }
085    
086    private void doReleaseConnection(ConnectionAdapter conn)
087    {
088        Class<?>[] types =
089        { Connection.class, DataSource.class };
090        Method m;
091        try
092        {
093            //DataSourceUtils.doReleaseConnection(conn, dsProxy.getTargetDataSource());
094            DataSource dataSourceTarget = null;
095            m = classForName(SPRING_DATASOURCE_UTILS).getDeclaredMethod("doReleaseConnection", types);
096            m.invoke(null, new Object[]
097            { (Connection)conn.unwrap(), dataSourceTarget });
098        }
099        catch (Exception e)
100        // TODO design exception IllegalArgumentException, IllegalAccessException, InvocationTargetException, SecurityException, NoSuchMethodException
101        {
102            handlerException.handle(e, "Cannot invoke [doReleaseConnection] from " + SPRING_DATASOURCE_UTILS + " class");
103        }
104    }
105    
106    private Class<?> classForName(String name)
107    {
108        try
109        {
110            return Class.forName(name);
111        }
112        catch (ClassNotFoundException e)
113        {
114            throw new RepositoryException("Cannot returns the Class object associated with the class [" + name + "]",
115                    e);
116        }
117    }
118}
