046/**
047 * 
048 * Inject the ResultSet at flat Object, basic types like: String, Date, Numbers, Clob, Blob...
049 * <p>
050 * <strong>This class doesn't supports inject value at Oriented-Object model, like nested objects.</strong>
051 * 
052 * @author Alisson Gomes
053 *
054 * @param <T> generic type of {@code Class} object to inject value of <code>ResultSet</code>
055 */
056public class PojoResultRow<T> implements ResultRow<T, ResultSet>
057{
058    private final static Logger          LOG = LoggerFactory.getLogger(PojoResultRow.class);
059    private final static MethodName SETTER =  MethodNameFactory.getInstanceSetter();
060    private final static MethodName GETTER =  MethodNameFactory.getInstanceGetter();
061    private final SqlLogger              sqlLogger;
062    private final Class<T>               returnType;
063    private final Set<OneToMany>         oneToManies;
064    private final Transformable<T> transformable;
065    private JdbcColumn<ResultSet>[]           columns;
066    
067    @SuppressWarnings("unchecked")
068    public PojoResultRow(Class<T> returnType, JdbcColumn<ResultSet>[] columns, Set<OneToMany> oneToManies, SqlLogger log)
069    {
070        this.returnType = returnType;
071        this.columns = columns;
072        this.oneToManies = oneToManies;
073        this.sqlLogger = log;
074        this.transformable = (Transformable<T>) new ObjectTransform();
075    }
076    
077    public T row(ResultSet rs, int rownum) throws SQLException
078    {
079        final Map<OneToMany, Object> otmValues;
080        otmValues = new HashMap<OneToMany, Object>();
081        ObjectProxy<T> proxyRow = ObjectProxyFactory.newProxy(returnType);
082        for (JdbcColumn<ResultSet> column : columns)
083        {
084            OneToMany otm = getOneToMany(column, otmValues);
085            if (otm == null)
086                setValueOf(column, rs, proxyRow);
087            else
088            {
089                prepareOneToManyValue(otm, column, rs, otmValues);
090            }
091        }
092        for (Entry<OneToMany, Object> entry : otmValues.entrySet())
093        {
094         
095            String attrName = entry.getKey().getProperty();
096            String getterName = GETTER.capitalize(attrName);
097            String setterName = SETTER.capitalize(attrName);
098            Collection<Object> collection = (Collection<Object>) proxyRow.invoke(getterName);
099            if (collection ==  null)
100            {
101                collection = (Collection<Object>) ObjectProxyFactory.newProxy(entry.getKey().getImpl()).newInstance();
102                proxyRow.invoke(setterName, collection);
103            }
104            //System.out.println("adding to collection: " + entry.getValue());
105            collection.add(entry.getValue());
106            //proxyRow.invoke(SETTER.capitalize(entry.getKey().getProperty()), entry.getValue());
107        }
108        //System.out.println(proxyRow.getInstance());
109        return proxyRow.getInstance();
110    }
111    
112    private void setValueOf(JdbcColumn<ResultSet> column, ResultSet rs, ObjectProxy<T> proxy) throws SQLException
113    {
114        Injectable<T> reflect = InjectableFactory.newMethodInjection(proxy);
115        Object jdbcObject = null;
116        if (column.isBinary())
117            jdbcObject = column.getBytes(rs);
118        else
119            jdbcObject = column.getValue(rs);
120
121        if (column.isNestedAttribute())
122            reflect.inject(column.getAttributeName(), jdbcObject);
123        else
124        {
125            String method = column.getMethodName();
126            if (proxy.hasMethod(method))
127                reflect.inject(method, jdbcObject);
128            else
129                LOG.info("Method [{}] doesn't exists for [{}] to set value [{}]", method,
130                        proxy.getTargetClass().getName(), jdbcObject);
131        }
132    }
133    
134    
135    private OneToMany getOneToMany(JdbcColumn<ResultSet> jdbcColumn, final Map<OneToMany, Object> otmValues)
136    {
137        OneToMany otm = null;
138        for (OneToMany m : oneToManies)
139        {
140            if (jdbcColumn.getName().startsWith(m.getProperty() + "."))
141            {
142                otm = m;
143                if (!otmValues.containsKey(otm))
144                {
145                    ObjectProxy<?> proxy = ObjectProxyFactory.newProxy(otm.getTypeOf());
146                    otmValues.put(otm, proxy.newInstance());
147                }
148                break;
149            }
150        }
151        return otm;
152    }
153    
154    private void prepareOneToManyValue(OneToMany otm, JdbcColumn<ResultSet> column, ResultSet rs, final Map<OneToMany, Object> otmValues) throws SQLException
155    {
156        ObjectProxy<?> proxy = ObjectProxyFactory.newProxy(otmValues.get(otm));
157        Injectable<?> reflect = InjectableFactory.newMethodInjection(proxy);
158        Object jdbcObject = null;
159        if (column.isBinary())
160            jdbcObject = column.getBytes(rs);
161        else
162            jdbcObject = column.getValue(rs);
163        // otm.property : 'book', JdbcColumn: book.name, capitalize -> setName
164        String method = SETTER.capitalize(column.getName().substring(otm.getProperty().length()+1));
165        reflect.inject(method, jdbcObject);
166    }
