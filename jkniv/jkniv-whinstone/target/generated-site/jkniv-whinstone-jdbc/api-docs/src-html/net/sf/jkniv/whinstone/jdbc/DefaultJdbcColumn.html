020package net.sf.jkniv.whinstone.jdbc;
021
022import java.sql.ResultSet;
023import java.sql.SQLException;
024import java.sql.Types;
025import java.util.Date;
026import java.util.Locale;
027
028import net.sf.jkniv.sqlegance.JdbcColumn;
029import net.sf.jkniv.sqlegance.JdbcColumnMapper;
030import net.sf.jkniv.sqlegance.UnderscoreToCamelCaseMapper;
031
032public class DefaultJdbcColumn implements JdbcColumn<ResultSet>
033{
034    private final int                     columnIndex;
035    private final String                  columnName;
036    private final String                  attributeName;
037    private final String                  methodName;
038    private final int                     jdbcType;
039    private boolean nestedAttribute;
040    private final static JdbcColumnMapper jdbcColumnMapper = new UnderscoreToCamelCaseMapper();// TODO design property to config;
041    
042    public DefaultJdbcColumn(int columnIndex, String columnName, int jdbcType)
043    {
044        super();
045        this.columnIndex = columnIndex;
046        this.columnName = columnName;
047        this.nestedAttribute = false;
048        if(columnName.indexOf(".") > 0)
049        {
050            this.nestedAttribute = true;
051            this.methodName = columnName;
052            this.attributeName = columnName;
053        }
054        else
055        {
056            this.attributeName = jdbcColumnMapper.map(columnName);
057            this.methodName = capitalizeSetter(attributeName);
058        }
059        this.jdbcType = jdbcType;
060        //this.jdbcColumnMapper = new UnderscoreToCamelCaseMapper();// TODO design property to config
061    }
062    
063    public String getAttributeName()
064    {
065        return this.attributeName;
066    }
067    
068    public String getMethodName()
069    {
070        return methodName;
071    }
072    
073    public String getName()
074    {
075        return this.columnName;
076    }
077    
078    public int getIndex()
079    {
080        return this.columnIndex;
081    }
082    
083    public boolean isBinary()
084    {
085        return (this.jdbcType == Types.CLOB || jdbcType == Types.BLOB);
086    }
087    
088    public boolean isClob()
089    {
090        // FIXME implements read CLOB as string, 
091        // FIXME implements write CLOB to database
092        return (this.jdbcType == Types.CLOB);
093    }
094    
095    public boolean isBlob()
096    {
097        // FIXME implements write BLOB to database
098        return (this.jdbcType == Types.BLOB);
099    }
100    
101    public boolean isDate()
102    {
103        return (this.jdbcType == Types.DATE);
104    }
105    
106    public boolean isTimestamp()
107    {
108        return (this.jdbcType == Types.TIMESTAMP);
109    }
110    
111    public boolean isTime()
112    {
113        return (this.jdbcType == Types.TIME);
114    }
115    
116    public boolean isNestedAttribute()
117    {
118        return nestedAttribute;
119    }
120    
121    @Override
122    public Object getValue(ResultSet rs) throws SQLException
123    {
124        Object value = rs.getObject(columnIndex);
125        if (value != null)
126        {
127            if (isDate())
128                value = new Date(rs.getDate(columnIndex).getTime());
129            else if (isTimestamp())
130                value = new Date(rs.getTimestamp(columnIndex).getTime());
131            else if (isTime())
132                value = new Date(rs.getTime(columnIndex).getTime());
133        }
134        return value;
135    }
136    
137    @Override
138    public Object getBytes(ResultSet rs) throws SQLException
139    {
140        return rs.getBytes(columnIndex);
141    }
142    
143    /**
144     * Append prefix <code>set<code> to attributeColumnName and capitalize it.
145     * @param attributeColumnName attribute name to capitalize with <code>set</code> prefix
146     * @return return capitalize attribute name, sample: identityName -> setIdentityName
147     */
148    private String capitalizeSetter(String attributeColumnName)// TODO design config capitalize algorithm
149    {
150        String capitalize = "";
151        
152        if (attributeColumnName != null)
153        {
154            int length = attributeColumnName.length();
155            capitalize = attributeColumnName.substring(0, 1).toUpperCase(Locale.ENGLISH);
156            if (length > 1)
157                capitalize += attributeColumnName.substring(1, length);
158        }
159        //sqlLogger.log(LogLevel.RESULTSET, "Mapping column [{}] to property [{}]", attributeColumnName, "set" + capitalize);
160        return "set" + capitalize;
161    }
162
163    @Override
164    public String toString()
165    {
166        return "DefaultJdbcColumn [index=" + columnIndex + ", columnName=" + columnName + ", jdbcType=" + jdbcType
167                + "]";
168    }
169}
