001package net.sf.jkniv.whinstone.jdbc;
002
003import net.sf.jkniv.sqlegance.RepositoryException;
004
005public class UnsupportedTransactionException extends RepositoryException
006{

008    
009    
010    /**
011     * Constructor for UnsupportedTransactionException.
012     * @param msg the detail from exception message
013     */
014    public UnsupportedTransactionException(String msg) {
015        super(msg);
016    }
017
018    /**
019     * Constructor for UnsupportedTransactionException.
020     * @param msg the detail from exception message
021     * @param cause the trouble root cause , usually JDBC family exception
022     */
023    public UnsupportedTransactionException(String msg, Throwable cause) {
024        super(msg, cause);
025    }
026}
