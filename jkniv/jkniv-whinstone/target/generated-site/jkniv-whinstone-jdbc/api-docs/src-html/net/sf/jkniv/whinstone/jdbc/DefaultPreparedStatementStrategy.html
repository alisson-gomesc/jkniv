075    /**
076     * Creates a PreparedStatement object that will generate ResultSet objects with the given type, concurrency, and holdability.
077     * The parameters values is setting
078     * @param conn Opened connection to database
079     * @return a new PreparedStatement object, containing the pre-compiled SQL statement.
080     * @throws net.sf.jkniv.sqlegance.RepositoryException wrapper SQLException
081     * @see java.sql.SQLException
082     */
083    public PreparedStatement prepareStatement(Connection conn)
084    {
085        PreparedStatement stmt = null;
086        int rsType = ResultSet.TYPE_FORWARD_ONLY;
087        int rsConcurrency = ResultSet.CONCUR_READ_ONLY;
088        int rsHoldability = ResultSet.CLOSE_CURSORS_AT_COMMIT;
089        Sql isql = queryable.getSql();
090        if (isql.getResultSetType() == ResultSetType.TYPE_SCROLL_INSENSITIVE)
091            rsType = ResultSet.TYPE_SCROLL_INSENSITIVE;
092        else if (isql.getResultSetType() == ResultSetType.TYPE_SCROLL_SENSITIVE)
093            rsType = ResultSet.TYPE_SCROLL_SENSITIVE;
094        
095        if (isql.getResultSetConcurrency() == ResultSetConcurrency.CONCUR_UPDATABLE)
096            rsConcurrency = ResultSet.CONCUR_UPDATABLE;
097        if (isql.getResultSetHoldability() == ResultSetHoldability.HOLD_CURSORS_OVER_COMMIT)
098            rsHoldability = ResultSet.HOLD_CURSORS_OVER_COMMIT;
099        
100        try
101        {
102            if(LOG.isTraceEnabled())
103                LOG.trace("Preparing SQL statement type [{}], concurrency [{}], holdability [{}] with [{}] parameters", isql.getResultSetType(), isql.getResultSetConcurrency(), isql.getResultSetHoldability(), paramsNames.length);
104
105            if (queryable.getSql().getSqlDialect().supportsStmtHoldability())
106                stmt = conn.prepareStatement(sql, rsType, rsConcurrency, rsHoldability);
107            else 
108            {
109                // SQLServer doesn't support Holdability
110                stmt = conn.prepareStatement(sql, rsType, rsConcurrency);
111                conn.setHoldability(rsHoldability);
112            }
113                
114            //if (paramsNames.length > 0)
115            //    setValues(stmt);
116            
117            if (isql.getTimeout() > 0)
118                stmt.setQueryTimeout(isql.getTimeout());
119        }
120        catch (SQLException sqle)
121        {
122            throw new RepositoryException("Cannot prepare statement ["+sqle.getMessage()+"]", sqle);
123        }
124        return stmt;
125    }
126
127    /**
128     * Creates a {@code PreparedStatement} object object capable of returning the auto-generated 
129     * keys designated by the given array.
130     * 
131     * @param conn Opened connection to database
132     * @param columnNames an array of column names indicating the columns that should be returned from the inserted row or rows 
133     * @return a new PreparedStatement object, containing the pre-compiled SQL statement.
134     * @throws net.sf.jkniv.sqlegance.RepositoryException wrapper SQLException
135     * @see java.sql.SQLException
136     */
137    public PreparedStatement prepareStatement(Connection conn, String[] columnNames)
138    {
139        PreparedStatement stmt = null;
140        Sql isql = queryable.getSql();
141        try
142        {
143            if(LOG.isTraceEnabled())
144                LOG.trace("Preparing SQL statement type with [{}] column names", paramsNames.length);
145            
146            if (columnNames.length > 0)
147                stmt = conn.prepareStatement(sql, columnNames);
148            else
149                stmt = conn.prepareStatement(sql, Statement.RETURN_GENERATED_KEYS);
150            
151            //if (paramsNames.length > 0)
152            //    setValues(stmt);
153            
154            if (isql.getTimeout() > 0)
155                stmt.setQueryTimeout(isql.getTimeout());
156        }
157        catch (SQLException sqle)
158        {
159            throw new RepositoryException("Cannot prepare statement!", sqle);
160        }
161        return stmt;
162    }

164    public PreparedStatement prepareStatementCount(Connection conn)
165    {
166        PreparedStatement stmt = null;
167        try
168        {
169            stmt = conn.prepareStatement(sqlCount);
170            //if (paramsNames.length > 0)
171            //    setValues(stmt);
172        }
173        catch (SQLException sqle)
174        {
175            throw new RepositoryException(
176                    "Cannot prepare count statement to [" + queryable.getName() + "]!", sqle);
177        }
178        return stmt;
179    }
