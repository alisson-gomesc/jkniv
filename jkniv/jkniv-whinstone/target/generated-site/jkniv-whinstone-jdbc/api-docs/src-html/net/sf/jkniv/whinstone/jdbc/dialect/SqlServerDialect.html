085    @Override

087    {
088        //return "select TOP %2$s %1$s";
089        //return "%1$s OFFSET %2$s ROWS FETCH NEXT %3$s ROWS ONLY";
090        final StringBuilder pagingSelect = new StringBuilder(100);
091        pagingSelect.append(" WITH query AS (");
092        pagingSelect.append("  SELECT inner_query.*");
093        pagingSelect.append("      , ROW_NUMBER() OVER (ORDER BY CURRENT_TIMESTAMP) as _jkniv_rownum_");
094        pagingSelect.append("  FROM ( %1$s ) inner_query");
095        pagingSelect.append(" )");
096        pagingSelect.append(" SELECT * FROM query WHERE _jkniv_rownum_ > %2$s AND _jkniv_rownum_ <= %3$s + %2$s");
097        return pagingSelect.toString();
098
099    }

164    /**
165     * Returns index of the first case-insensitive match of search term surrounded by spaces
166     * that is not enclosed in parentheses.
167     *
168     * @param sb String to search.
169     * @param search Search term.
170     * @param fromIndex The index from which to start the search.
171     *
172     * @return Position of the first match, or {@literal -1} if not found.
173     */
174    private static int shallowIndexOfWord(final StringBuilder sb, final String search, int fromIndex) {
175        final int index = shallowIndexOf( sb, ' ' + search + ' ', fromIndex );
176        // In case of match adding one because of space placed in front of search term.
177        return index != -1 ? ( index + 1 ) : -1;
178    }
179
180    /**
181     * Returns index of the first case-insensitive match of search term that is not enclosed in parentheses.
182     *
183     * @param sb String to search.
184     * @param search Search term.
185     * @param fromIndex The index from which to start the search.
186     *
187     * @return Position of the first match, or {@literal -1} if not found.
188     */
189    private static int shallowIndexOf(StringBuilder sb, String search, int fromIndex) {
190        // case-insensitive match
191        final String lowercase = sb.toString().toLowerCase(Locale.ROOT);
192        final int len = lowercase.length();
193        final int searchlen = search.length();
194        int pos = -1;
195        int depth = 0;
196        int cur = fromIndex;
197        do {
198            pos = lowercase.indexOf( search, cur );
199            if ( pos != -1 ) {
200                for ( int iter = cur; iter < pos; iter++ ) {
201                    final char c = sb.charAt( iter );
202                    if ( c == '(' ) {
203                        depth = depth + 1;
204                    }
205                    else if ( c == ')' ) {
206                        depth = depth - 1;
207                    }
208                }
209                cur = pos + searchlen;
210            }
211        } while ( cur < len && depth != 0 && pos != -1 );
212        return depth == 0 ? pos : -1;
213    }
