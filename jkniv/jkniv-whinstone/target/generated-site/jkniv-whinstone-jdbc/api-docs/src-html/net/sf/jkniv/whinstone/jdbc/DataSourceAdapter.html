020package net.sf.jkniv.whinstone.jdbc;
021
022import java.sql.Connection;
023
024import javax.sql.DataSource;
025
026import net.sf.jkniv.sqlegance.ConnectionAdapter;
027import net.sf.jkniv.sqlegance.transaction.Isolation;
028
029public class DataSourceAdapter extends AbstractJdbcAdapter
030{
031    private DataSource          dataSource;
032    private Isolation           defaultIsolation;
033    
034    public DataSourceAdapter(DataSource ds, String contextName)//DataSource dataSource, Isolation defaultIsolation, String name)
035    {
036        super(contextName);
037        this.dataSource = ds;
038        this.defaultIsolation = Isolation.DEFAULT;
039    }
040    
041    /*
042    public DataSourceConnection(RepositoryConfig config)//DataSource dataSource, Isolation defaultIsolation, String name)
043    {
044        super(config);
045        this.dataSource = config.getDataSource();
046        this.defaultIsolation = Isolation.DEFAULT;
047    }
048    */
049    
050    
051    /**
052     * Attempts to establish a connection to the database 
053     * @return a Connection from DataSource
054     * @throws net.sf.jkniv.sqlegance.RepositoryException if cannot establish a connection
055     */
056    public ConnectionAdapter open()
057    {
058        return open(defaultIsolation);
059    }
060    
061    
062    /**
063     * Attempts to establish a connection to the database with specific isolation 
064     * @param isolation transaction level for connection
065     * @return a Connection from DataSource
066     * @throws net.sf.jkniv.sqlegance.RepositoryException if cannot establish a connection
067     */
068    public ConnectionAdapter open(Isolation isolation)
069    {
070        ConnectionAdapter adapter = null;
071        try
072        {
073            LOG.debug("Getting Connection from DataSource");
074            Connection conn = dataSource.getConnection();
075            setIsolation(conn, isolation);
076            adapter = new JdbcConnectionAdapter(conn);
077        }
078        catch (Exception e)//SQLException
079        {
080            handlerException.handle(e, "SEVERE FAIL, cannot get database connection datasource [" + dataSource
081                    + "] Reason: " + e.getMessage());
082        }
083        return adapter;
084    }
085    
086}
