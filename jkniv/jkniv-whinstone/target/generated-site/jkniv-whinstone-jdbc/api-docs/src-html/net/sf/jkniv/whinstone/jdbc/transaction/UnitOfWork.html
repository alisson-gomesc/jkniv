102    @Override
103    public int execute(Queryable queryable)
104    {
105        ConnectionAdapter adapterConn = null;
106        Connection conn = null;
107        PreparedStatement stmt = null;
108        Integer affected = 0;
109        try
110        {
111            //SqlDialect sqlDialect = newDialect(queryable);
112            adapterConn = getConnection(queryable.getSql().getIsolation());
113            conn = (Connection)adapterConn.unwrap();
114            PreparedStatementStrategy stmtStrategy = new DefaultPreparedStatementStrategy(queryable, sqlLogger);//getPreparedStatementStrategy(sqlDialect);
115            
116            if (queryable.getSql().isInsertable())
117            {
118                DbCommand command = DbCommandFactory.newInstance(queryable, conn, stmtStrategy);
119                affected = command.execute();
120            }
121            else
122            {
123                DbCommand command = DbCommandFactory.newInstance(queryable, adapterConn, stmtStrategy);
124                affected = command.execute();
125//                StatementAdapter<Number, ResultSet> adapterStmt = adapterConn.newStatement(queryable);
126//                queryable.bind(adapterStmt).on();
127//                affected = adapterStmt.execute();
128            }
129        }
130        finally
131        {
132            connectionFactory.close(stmt);
133            connectionFactory.close(adapterConn);
134        }
135        return affected;
136    }

318    private ConnectionAdapter getConnection(Isolation isolation)
319    {
320        ConnectionAdapter conn = null;
321        TransactionContext transactionContext = TransactionSessions.get(connectionFactory.getContextName());
322        
323        if (transactionContext != null && transactionContext.isActive())
324        {
325            LOG.debug("Taking existent Connection from Transaction Context");
326            conn = transactionContext.getConnection();
327        }
328        if (conn == null)
329        {
330            conn = connectionFactory.open(isolation);
331        }
332        return conn;
333    }

352    private <T> void setParseRow(Queryable queryable, Sql isql, Class<T> returnType, JdbcColumn[] columns)
353    {
354        ResultRow<T, ?> rsRowParser = null;
355        //JdbcColumn[] columns = JdbcFactory.getJdbcColumns(rs.getMetaData());
356        Selectable selectTag = (Selectable) isql;
357        if (queryable.isScalar())
358        {
359            rsRowParser = new ScalarResultRow(columns, sqlLogger);
360        }
361        else if (Map.class.isAssignableFrom(returnType))
362        {
363            rsRowParser = new MapResultRow(returnType, columns, sqlLogger);
364            //transformable = new MapTransform();
365        }
366        else if (Number.class.isAssignableFrom(returnType)) // FIXME implements for date, calendar, boolean improve design
367        {
368            rsRowParser = new NumberResultRow(returnType, columns, sqlLogger);
369        }
370        else if (String.class.isAssignableFrom(returnType))
371        {
372            rsRowParser = new StringResultRow(columns, sqlLogger);
373        }
374        else if (selectTag.getOneToMany().isEmpty())
375        {
376            rsRowParser = new FlatObjectResultRow(returnType, columns, sqlLogger);
377            //transformable = new ObjectTransform();
378        }
379        else
380        {
381            rsRowParser = new PojoResultRow(returnType, columns, selectTag.getOneToMany(), sqlLogger);
382            //transformable = new ObjectTransform();
383        }
