020package net.sf.jkniv.whinstone.jdbc.result;
021
022import java.sql.ResultSet;
023import java.sql.SQLException;
024
025import org.slf4j.Logger;
026import org.slf4j.LoggerFactory;
027
028import net.sf.jkniv.reflect.Injectable;
029import net.sf.jkniv.reflect.InjectableFactory;
030import net.sf.jkniv.reflect.beans.ObjectProxy;
031import net.sf.jkniv.reflect.beans.ObjectProxyFactory;
032import net.sf.jkniv.sqlegance.JdbcColumn;
033import net.sf.jkniv.sqlegance.ResultRow;
034import net.sf.jkniv.sqlegance.classification.ObjectTransform;
035import net.sf.jkniv.sqlegance.classification.Transformable;
036import net.sf.jkniv.sqlegance.logger.SqlLogger;
037
038/**
039 * 
040 * Inject the ResultSet at flat Object, basic types like: String, Date, Numbers, Clob, Blob...
041 * <p>
042 * <strong>This class doesn't supports inject value at Oriented-Object model, like nested objects.</strong>
043 * 
044 * @author Alisson Gomes
045 *
046 * @param <T> generic type of {@code Class} object to inject value of <code>ResultSet</code>
047 */
048public class FlatObjectResultRow<T> implements ResultRow<T, ResultSet>
049{
050    private final static Logger LOG = LoggerFactory.getLogger(FlatObjectResultRow.class);
051    //private final SqlLogger    sqlLogger;
052    private final Class<T>     returnType;
053    private final Transformable<T> transformable;
054    private JdbcColumn<ResultSet>[] columns;
055    
056    @SuppressWarnings("unchecked")
057    public FlatObjectResultRow(Class<T> returnType, JdbcColumn<ResultSet>[] columns, SqlLogger log)
058    {
059        this.returnType = returnType;
060        //this.sqlLogger = log;
061        this.columns = columns;
062        this.transformable = (Transformable<T>) new ObjectTransform();
063    }
064    
065    public T row(ResultSet rs, int rownum) throws SQLException
066    {
067        ObjectProxy<T> proxy = ObjectProxyFactory.newProxy(returnType);
068        for (JdbcColumn<ResultSet> column : columns)
069            setValueOf(column, rs, proxy);
070        
071        return proxy.getInstance();
072    }
073    
074    private void setValueOf(JdbcColumn column, ResultSet rs, ObjectProxy<T> proxy) throws SQLException
075    {
076        Injectable<T> reflect = InjectableFactory.newMethodInjection(proxy);
077        Object jdbcObject = null;
078        if (column.isBinary())
079            jdbcObject = column.getBytes(rs);
080        else
081            jdbcObject = column.getValue(rs);
082        
083        if (column.isNestedAttribute())
084            reflect.inject(column.getAttributeName(), jdbcObject);
085        else
086        {
087            String method = column.getMethodName();
088            if (proxy.hasMethod(method))
089                reflect.inject(method, jdbcObject);
090            else
091                LOG.warn("Method [{}] doesn't exists for [{}] to set value [{}]", method, proxy.getTargetClass().getName(), jdbcObject);
092        }
093    }
094
095    @Override
096    public Transformable<T> getTransformable()
097    {
098        return transformable;
099    }
100    
101    @Override
102    public void setColumns(JdbcColumn<ResultSet>[] columns)
103    {
104        this.columns = columns;
105    }
106
107}
